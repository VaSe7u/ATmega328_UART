33 void UART_Init(const UART_CharacterSize characterSize,
34  const UART_Parity parity,
35  const UART_StopBits stopBits) {
36  // Configure baud rate
37  #include <util/setbaud.h>
38  UBRR0H = UBRRH_VALUE;
39  UBRR0L = UBRRL_VALUE;
40 
41  #if USE_2X
42  UCSR0A |= (1 << U2X0);
43  #else
44  UCSR0A &= ~(1 << U2X0);
45  #endif
46 
47  UCSR0A &= ~(1 << FE0); // clear framing error
48 
49  // Enable RX, TX and partially configure character size
50  UCSR0B = ((1 << RXEN0) | (1 << TXEN0) | (characterSize & 0x04));
51 
52  // Configure parity, stop bits and character size
53  UCSR0C = (parity | stopBits | ((characterSize & 0x03) << UCSZ00));
54 }

77 void UART_Putchar(const uint8_t data, FILE *stream) {
78  // Append line feed to newline character
79  if (data == '\n') {
80  UART_Putchar('\r', 0);
81  }
82  // Wait until write buffer is empty
83  while ((UCSR0A & (1 << UDRE0)) == 0) {}
84  UDR0 = data;
85 }
